-----------------------------------
Redis configuration file is usually named redis.windows.conf or redis.conf.

You can:

Change the port (default 6379)

Enable password (via requirepass <password>)

Enable persistence (AOF or RDB)


------------------------------------
Add Redis client dependency (e.g. Jedis or Lettuce) to your Maven project:

<dependency>
  <groupId>redis.clients</groupId>
  <artifactId>jedis</artifactId>
  <version>5.1.0</version>
</dependency>

------------------------------------
import redis.clients.jedis.Jedis;

public class RedisExample {
    public static void main(String[] args) {
        try (Jedis jedis = new Jedis("localhost", 6379)) {
            jedis.auth("myStrongPassword"); // if you set password
            jedis.set("greeting", "Hello Redis!");
            System.out.println(jedis.get("greeting"));
        }
    }
}
